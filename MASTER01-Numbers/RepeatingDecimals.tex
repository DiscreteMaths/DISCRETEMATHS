\begin{frame}
\frametitle{Repeating Decimals}
\Large
\begin{itemize}
\item Let $x = 0.6363....$
\item Multiply by 10 \[10x = 6.3636..\]
\item Multiply by 100 \[100x = 63.6363...\]
\end{itemize}
\end{frame}
%---------------------------------------------%
\begin{frame}
\frametitle{Repeating Decimals}
\Large
\begin{itemize}
\item (Last Slide : $100x = 63.6363...$)
\item Remember $x = 0.6363....$ 
\[100x = 63 + x\]
\item Re-arrange algebraically
\[99x = 63\]
\item Divide both sides
\[ x = 63/99\]
\end{itemize}
