\frametitle{Binary Addition}
\Large
\vspace{-2cm}
\textbf{Rules of Binary Addition}
\begin{itemize}
\item 0 + 0 = 0
\item 0 + 1 = 1
\item 1 + 0 = 1
\item 1 + 1 = 0 \\ (and carry 1 to the next more significant bit)
\item 1 + 1 + 1 = 1 \\ (and carry 1 to the next more significant bit)
\end{itemize}
\end{frame}
